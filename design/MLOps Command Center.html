Plotly dark theme

Drop this layout dict into every fig.update_layout(**MC_PLOTLY). Cyan-led monochrome; amber reserved for threshold lines only.

# mc_plotly.py — shared chart theme
MC_PLOTLY = dict(
  paper_bgcolor="#111827", plot_bgcolor="#0b0f1a",
  font=dict(family="JetBrains Mono, monospace", size=11, color="#64748b"),
  colorway=["#00d4ff", "#5ce6ff", "#0891b2", "#7de8ff"],  # cyan monochrome
  margin=dict(l=48, r=16, t=16, b=32),
  xaxis=dict(gridcolor="#1a2740", zerolinecolor="#243456",
             linecolor="#243456", tickcolor="#243456"),
  yaxis=dict(gridcolor="#1a2740", zerolinecolor="#243456",
             linecolor="#243456", tickcolor="#243456"),
  hoverlabel=dict(bgcolor="#161f33", bordercolor="#00d4ff",
                  font=dict(color="#e2e8f0", family="JetBrains Mono")),
  legend=dict(bgcolor="rgba(0,0,0,0)", font=dict(color="#a9b6cc")),
)
# threshold lines: fig.add_hline(y=thr, line_dash="dash", line_color="#f59e0b")
# line traces: line=dict(width=2, color="#00d4ff"); fill="tozeroy", fillcolor="rgba(0,212,255,0.12)"
